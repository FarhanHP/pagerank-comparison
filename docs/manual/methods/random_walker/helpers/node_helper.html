from methods.random_walker.helper_factories.forward_links_helper_factory import ForwardLinksHelperFactory
from methods.random_walker.helpers.forward_links_helper import ForwardLinksHelper
from methods.random_walker.helpers.nodes_helper import NodesHelper
from methods.random_walker.helpers.page_informations_helper import PageInformationsHelper
from model import Node
from data.page_linking_repository import PageLinkingRepository


class NodeHelper:
  """
  Properties:
    - page_linking_repository: object to access page_linking table entries in the database
    - nodes_helper: helper class to manage all nodes in this program
    - page_informations_helper: helper class to manage all page informations in this program
    - forward_links_helper_factory: factory class to construct ForwardLinkHelper
    - damping_factor
    - node: node object this helper manage
    - forward_links_helper: helper class to manage forward links that the node has, will be constructed lazily by forward_links_helper_factory
  """

  __page_linking_repository: PageLinkingRepository

  __nodes_helper: NodesHelper
  __page_informations_helper: PageInformationsHelper

  __forward_links_helper_factory: ForwardLinksHelperFactory
  
  __damping_factor: float
  __node: Node
  __forward_links_helper: ForwardLinksHelper | None = None
 
  def insert_stateful_params(self, node: Node, nodes_helper: NodesHelper, page_informations_helper: PageInformationsHelper):
    """
    method to insert parameters that are stateful. Stateful means for every instances will have different mutable object params

    Args:
      - node: Node object this helper manage
      - nodes_helper: helper object managing all nodes this program has
      - page_informations_helper: helper object managing all page informations this program has
    """

    self.__node = node
    self.__nodes_helper = nodes_helper
    self.__page_informations_helper = page_informations_helper

  def __init__(
    self, 
    damping_factor: float, 
    page_linking_repository: PageLinkingRepository, 
    forward_links_helper_factory: ForwardLinksHelperFactory
  ) -> None:
    self.__damping_factor = damping_factor
    self.__forward_links_helper_factory = forward_links_helper_factory
    self.__page_linking_repository = page_linking_repository

  def get_destination_nodes(self) -> list[Node]:
    """
    get all destination nodes base on this helper's node forward links 
    """

    return self.__get_forward_links_helper().get_destination_nodes()
  
  def get_destination_probabilites(self) -> list[float]:
    """
    get all destination probabilites base on this helper's node forward links 
    """

    return self.__get_forward_links_helper().get_probabilites()
  
  def __get_forward_links_helper(self) -> ForwardLinksHelper:
    if(self.__forward_links_helper is None):
      self.__forward_links_helper = self.__create_forward_links_helper()

    return self.__forward_links_helper
  
  def __create_forward_links_helper(self) -> ForwardLinksHelper:
    forward_links = self.__nodes_helper.create_initial_forward_links(self.__page_informations_helper.get_pages_count())
    forward_links_helper = self.__forward_links_helper_factory.create_helper(forward_links)

    page_information = self.__node.get_page_information()
    forward_linkings = self.__page_linking_repository.get_page_linkings_by_page_id(page_information.id_page)
    pages_count = self.__page_informations_helper.get_pages_count()
    cj = len(forward_linkings)
    initial_probability_value_when_cj_gt_0 = (1 - self.__damping_factor) / pages_count

    if(cj > 0):
      forward_links_helper.set_forward_links_probability(initial_probability_value_when_cj_gt_0)
    
    for forward_linking in forward_linkings:
      destination_node = self.__nodes_helper.get_node_by_url(forward_linking.outgoing_link)
      forward_link = forward_links_helper.get_forward_link_by_destination_node(destination_node)
      probability_value_when_there_is_connection_to_destionation_node = self.__damping_factor/cj + initial_probability_value_when_cj_gt_0
      forward_link.set_probability(probability_value_when_there_is_connection_to_destionation_node)

    return forward_links_helper
  
  def get_node(self) -> Node:
    """
    get this helper's node
    """

    return self.__node
